require("lualibs")
local f = io.open("problems-tex.json", "r")
if not f then
  tex.sprint("\\textbf{Error: problems-tex.json not found. Run convert.py first.}")
  return
end
local json_str = f:read("*all")
f:close()

local problems = utilities.json.tolua(json_str)

-- Escape all TeX special characters so descriptions render as plain text
local special_chars = {
  ["\\"] = "\\textbackslash{}",
  ["{"]  = "\\{",
  ["}"]  = "\\}",
  ["$"]  = "\\$",
  ["%"]  = "\\%",
  ["&"]  = "\\&",
  ["#"]  = "\\#",
  ["^"]  = "\\textasciicircum{}",
  ["_"]  = "\\_",
  ["~"]  = "\\textasciitilde{}",
}
local function safe_text(s)
  return (s:gsub("[\\{}$%%&#^_~]", special_chars))
end

for i, p in ipairs(problems) do
  local num_title = i .. ". " .. p.title
  local desc      = safe_text(p.description)
  local formula   = p.latex

  tex.sprint("\\medskip")
  tex.sprint("\\noindent{\\large\\bfseries\\mathversion{normal}\\color{titleblue}" .. num_title .. "}\\par")
  tex.sprint("\\noindent{\\itshape " .. desc .. "}\\par\\nobreak")
  tex.sprint("\\[" .. formula .. "\\]")
  tex.sprint("\\noindent\\rule{\\linewidth}{0.4pt}\\par")
end
\end{luacode*}
